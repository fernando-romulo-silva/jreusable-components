		if (!function.isActice()) {
			LOGGER.debug("Function {} disabled, it won't execute", functionName);
			skippedFunctions.add(functionName);
			continue;
		}

		if (function.reTrowException()) {
			throw ex;
		}
		withErrorFunctions.add(functionName);
